# WINTOP - Windows System Monitor
# Windows-compatible requirements

# Required - Core functionality
psutil>=5.9.0

# NVIDIA GPU Support (choose one or both)
# Recommended: Easy to install, works immediately
gputil>=1.4.0

# Advanced NVIDIA support with more metrics
nvidia-ml-py3>=7.352.0

# Windows Management Instrumentation (for AMD, Intel, Generic GPU support)
# Windows only - automatically detects AMD, Intel, and other GPUs
wmi>=1.5.1; sys_platform == 'win32'

# ========================================
# INSTALLATION GUIDE
# ========================================

# For NVIDIA GPU users:
# pip install psutil gputil wmi

# For AMD GPU users:
# pip install psutil wmi
# Note: AMD support uses WMI. For advanced metrics, install OpenHardwareMonitor

# For Intel GPU users:
# pip install psutil wmi

# For Multi-GPU systems:
# pip install psutil gputil nvidia-ml-py3 wmi

# Minimal (no GPU monitoring):
# pip install psutil

# ========================================
# NOTES
# ========================================
# - pyamdgpuinfo does NOT work on Windows (requires Linux libdrm)
# - AMD/Intel GPU metrics are limited on Windows without OpenHardwareMonitor
# - OpenHardwareMonitor (optional): Download from https://openhardwaremonitor.org/
#   Run it in background for enhanced GPU temperature/load monitoring
